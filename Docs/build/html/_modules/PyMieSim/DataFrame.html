[docs]    def Plot(self, y='Coupling', **kwargs):

        ax = self.unstack(level=[-3,-1]).plot(y       = y,
                                              grid    = True,
                                              figsize = (8,4),
                                              xlabel  = r'Scatterer diameter [m]',
                                              ylabel  = r'Coupling [u.a.]',
                                              **kwargs)

        ax.legend(prop={'size': 8})
        plt.show()
